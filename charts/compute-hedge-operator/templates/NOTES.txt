   # Option A: apply a Secret (see config/secret.example.yaml), then restart:
   kubectl -n {{ .Release.Namespace }} apply -f config/secret.yaml
   kubectl -n {{ .Release.Namespace }} rollout restart deploy/{{ include "chp.fullname" . }}

   # Option B: helm upgrade with credentials.create=true and --set credentials.ornnApiToken=...

   with no code change. Marketplace posts only when MARKET_API_URL is set AND
   MARKET_WRITE_ENABLED=true.
